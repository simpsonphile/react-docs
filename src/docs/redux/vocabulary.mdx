export default function reducer(state = {}, action = {}) {
  if (action.type === SET_PLAYERS_INFO) {
    // do reducer stuff
  }

  if (action.type === UPDATE_PLAYER_POSITION_CORDS && action.cords.x && action.cords.y) {
    return {
      ...state,
      playerCords: action.cords,
    }
  }

  default: return state;
}
